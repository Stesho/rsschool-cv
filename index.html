function pathFinder(maze){
        let temp = maze.split('').filter(el => 
                      el === '.' || el === 'W');
                        
        let n = Math.sqrt(temp.length);
        let arr = [];
        
        for(let i = 0; i < n; i++) {
        arr[i] = [];
        }
        
        for(let i = 0, k = 0; i < n; i++) {
        for(let j = 0; j < n; j++, k++) {
          if(temp[k] != '\n') {
            arr[i][j] = temp[k];
          }
        }
        }
        
        function getSiblings (x, y) {
        let siblings = [];
        
        if(arr[x+1] !== undefined) {
          siblings.push({x:x+1, y:y, val:arr[x+1][y]});
        }
        if(arr[x-1] !== undefined) {
          siblings.push({x:x-1, y:y, val:arr[x-1][y]});
        }
        if(arr[x][y+1] !== undefined) {
          siblings.push({x:x, y:y+1, val:arr[x][y+1]});
        }
        if(arr[x][y-1] !== undefined) {
          siblings.push({x:x, y:y-1, val:arr[x][y-1]});
        }
        
        return siblings.filter(el => el.val === '.');
        }
        
        let func = (x, y) => {
        arr[x][y] = 'x';
        let sib = getSiblings(x, y);
        
        if(sib.length > 0) {
          for(let i = 0; i < sib.length; i++) {
            let current = sib[i];
            let isSolved = current.x === n-1
                        && current.y === n-1;
        
            let notVisited = 
                arr[current.x][current.y] !== 'x'; 
        
            if(isSolved || (notVisited && 
                            func(current.x, current.y))){
                return true;
            }
          }
        }
        return false;
        }
        
        return func(0, 0);
        }
